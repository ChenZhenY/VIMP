148 void calculate_partial_V(){

153 // GH approximation
154 update_covariance();
155 updateGH();
156
158 _gauss_hermite.update_integrand(_func_Vmu);
159
160 _Vdmu = _gauss_hermite.Integrate();
161 _Vdmu = _precision * _Vdmu;
162
164 _gauss_hermite.update_integrand(_func_phi);
165 double avg_phi = _gauss_hermite.Integrate()(0, 0);
166
168 _gauss_hermite.update_integrand(_func_Vmumu);
169 _Vddmu = _gauss_hermite.Integrate();
170
171 _Vddmu.triangularView<Upper>() = (_precision * _Vddmu * _precision).triangularView<Upper>();
172 _Vddmu.triangularView<StrictlyLower>() = _Vddmu.triangularView<StrictlyUpper>().transpose();
173
174 _Vddmu.triangularView<Upper>() = (_Vddmu - _precision * avg_phi).triangularView<Upper>();
175 _Vddmu.triangularView<StrictlyLower>() = _Vddmu.triangularView<StrictlyUpper>().transpose();

186 void calculate_exact_partial_V(VectorXd mu_t, MatrixXd covariance_t){
187 _Vdmu.setZero();
188 _Vddmu.setZero();
189 update_covariance();
190
191 // helper vectors
192 VectorXd eps{_mu - mu_t};
193 MatrixXd tmp{MatrixXd::Zero(_dim, _dim)};
194 MatrixXd precision_t{covariance_t.inverse()};
195
196 // partial V / partial mu
197 _Vdmu = precision_t * eps;
198
199 // partial V^2 / partial mu*mu^T
200 // update tmp matrix
201 for (int i=0; i<_dim; i++){
202 for (int j=0; j<_dim; j++) {
203 for (int k=0; k<_dim; k++){
204 for (int l=0; l<_dim; l++){
205 tmp(i, j) += (_covariance(i, j)*_covariance(k, l) + _covariance(i,k)*_covariance(j,l) + _covariance(i,l)*_covariance(j,k))*precision_t(k,l);
206 }
207 }
208 }
209 }
210
211 _Vddmu = _precision * tmp * _precision - _precision * (precision_t*_covariance).trace();
212 _Vddmu = _Vddmu / 2;
213
214 }
