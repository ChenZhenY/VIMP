183 void calculate_exact_partial_V(VectorXd mu_t, MatrixXd covariance_t){
184 Vdmu.setZero();
185 Vddmu.setZero();
186 update_covariance();
187
188 // helper vectors
189 VectorXd eps{mu_ - mu_t};
190 MatrixXd tmp{MatrixXd::Zero(dim_, dim_)};
191 MatrixXd precision_t{covariance_t.inverse()};
192
193 // partial V / partial mu
194 Vdmu = precision_t * eps;
195
196 // partial V^2 / partial mu*mu^T
197 // update tmp matrix
198 for (int i=0; i<dim_; i++){
199 for (int j=0; j<dim_; j++) {
200 for (int k=0; k<dim_; k++){
201 for (int l=0; l<dim_; l++){
202 tmp(i, j) += (covariance_(i, j)*covariance_(k, l) + covariance_(i,k)*covariance_(j,l) + covariance_(i,l)*covariance_(j,k))*precision_t(k,l);
203 }
204 }
205 }
206 }
207
208 Vddmu = precision_ * tmp * precision_ - precision_ * (precision_t*covariance_).trace();
209 Vddmu = Vddmu / 2;
210
211 }
