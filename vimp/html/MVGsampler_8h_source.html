197 Gaussian_distribution(const Eigen::VectorXd& mean, const Eigen::MatrixXd& covariance):mean_(mean), Covariance_(covariance){
198 dimension_ = mean.size();
199 precision_ = Covariance_.inverse();

201 assert(eigenSolver.eigenvalues().minCoeff() > 0);

204 double probability(Eigen::VectorXd x) const{
205 double temp = (-(x-mean_).transpose()*precision_*(x-mean_))(0)/2.0;
206 return exp(temp) / sqrt(pow(2 * PI_, dimension_) * Covariance_.determinant());
